pipeline {
    agent { label 'ss' }

    environment {
        DOCKERHUB_REPO = "akblazeacademy/akblazerepo"
        IMAGE_TAG      = "${BUILD_NUMBER}"
        CONTAINER_NAME = "pipeline-nginx"
        PORT           = "30080"
    }

    stages {

        stage('Checkout') {
            steps {
                git url: "${myrepo}", branch: 'main'
            }
        }

        stage('Build Image') {
            steps {
                sh '''
                  docker build -t ${DOCKERHUB_REPO}:${IMAGE_TAG} .
                '''
            }
        }

        stage('Docker Hub Login') {
            steps {
                withCredentials([
                    usernamePassword(
                        credentialsId: 'dockerhub-akblaze',
                        usernameVariable: 'DH_USER',
                        passwordVariable: 'DH_TOKEN'
                    )
                ]) {
                    sh '''
                      echo "$DH_TOKEN" | docker login -u "$DH_USER" --password-stdin
                    '''
                }
            }
        }

        stage('Tag & Push Image') {
            steps {
                sh '''
                  # Push build-number tag
                  docker push ${DOCKERHUB_REPO}:${IMAGE_TAG}

                  # Tag as latest
                  docker tag ${DOCKERHUB_REPO}:${IMAGE_TAG} ${DOCKERHUB_REPO}:latest

                  # Push latest
                  docker push ${DOCKERHUB_REPO}:latest
                '''
            }
        }

        stage('Run Container') {
            steps {
                sh '''
                  if docker ps -a --format '{{.Names}}' | grep -w ${CONTAINER_NAME}; then
                    docker rm -f ${CONTAINER_NAME}
                  fi

                  docker run -d \
                    -p ${PORT}:80 \
                    --name ${CONTAINER_NAME} \
                    ${DOCKERHUB_REPO}:latest
                '''
            }
        }
    }

    post {
        success {
            echo "‚úÖ Image pushed:"
            echo "   - ${DOCKERHUB_REPO}:${IMAGE_TAG}"
            echo "   - ${DOCKERHUB_REPO}:latest"
            echo "üöÄ App running on port ${PORT}"
        }
        failure {
            echo "‚ùå Pipeline failed"
        }
    }
}
